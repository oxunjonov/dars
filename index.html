📐 1NF — Birinchi Normal Shakl
Har bir ustun atomik (bo'linmas) qiymatlar saqlaydi
Takrorlanuvchi guruhlar yo'q (masalan, telefon1, telefon2 emas — alohida jadval)
Har bir satrda bitta asosiy kalit (PK) mavjud
Barcha ustunlar bir xil turdagi qiymatlar saqlaydi
Misol: TALABA jadvalida telefon alohida ustun (VARCHAR), massiv emas
📐 2NF — Ikkinchi Normal Shakl
1NF shartlari bajarilgan
Har bir atribut to'liq PK ga bog'liq (qisman bog'liqlik yo'q)
Kompozit PK bo'lsa — qisman bog'liqliklar alohida jadvalga chiqarilgan
BAHO jadvalida baho → talaba_id va fan_id ga bog'liq, alohida emas
Misol: DAVOMAT jadvalida UQ(talaba_id, jadval_id, sana) — kompozit kalit
📐 3NF — Uchinchi Normal Shakl
2NF shartlari bajarilgan
Tranzitiv bog'liqlik yo'q (A→B→C bo'lsa, C alohida jadvalga)
PK bo'lmagan ustunlar faqat PKga bog'liq
Fakultet nomi → Kafedra → Yo'nalish: har biri alohida jadvalda
Misol: TALABA jadvalida fakultet_nomi yo'q — FK orqali FAKULTET jadvaliga bog'langan
🔑 Kalit Turlari
PK (Primary Key) — har bir jadvalda bitta, NULL bo'lmaydi, AUTO_INCREMENT
FK (Foreign Key) — boshqa jadvalning PKga ishora, referensial yaxlitlik
UQ (Unique Key) — takrorlanmaydigan qiymat (pasport, email)
Kompozit PK — ikki va undan ko'p ustundan tashkil topgan kalit
🔗 Munosabat Turlari
1:1 — Talaba ↔ DiplomIshi (har biriga bitta)
1:N — Fakultet → Kafedra (eng ko'p ishlatiladigan)
N:M — Talaba ↔ Fan (BAHO jadvali orqali hal qilinadi)
N:M munosabat bog'lovchi jadval orqali 2 ta 1:N ga aylantiriladi
⚠️ Yaxlitlik Cheklovlari
NOT NULL — majburiy to'ldiriladigan ustunlar
UNIQUE — takrorlanmaydigan: pasport, email, talaba_raqam
CHECK — qiymat diapazoni: baho 0-100 orasida
DEFAULT — standart qiymat: holati = 'aktiv'
CASCADE — ota o'chirilsa, bolalar ham o'chadi (yoki SET NULL)

-- ================================================ -- FDTU TALABALAR AXBOROT TIZIMI -- Farg'ona Davlat Texnika Universiteti -- Relyatsion Model | 3NF | 17 ta jadval -- ================================================ CREATE DATABASE IF NOT EXISTS fdtu_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci; USE fdtu_db; -- 1. BINO CREATE TABLE bino ( bino_id INT AUTO_INCREMENT PRIMARY KEY, nomi VARCHAR(100) NOT NULL, manzil TEXT, qavatlar_soni TINYINT, turi ENUM('oquv','laboratoriya','yotoqxona','idora') ); -- 2. XONA CREATE TABLE xona ( xona_id INT AUTO_INCREMENT PRIMARY KEY, raqami VARCHAR(20) NOT NULL, bino_id INT NOT NULL, qavat TINYINT, sig_imkoniyat INT, turi ENUM('auditoriya','laboratoriya','konferens'), jihozlar TEXT, FOREIGN KEY (bino_id) REFERENCES bino(bino_id) ); -- 3. FAKULTET (dekan_id keyinroq ALTER bilan) CREATE TABLE fakultet ( fakultet_id INT AUTO_INCREMENT PRIMARY KEY, nomi VARCHAR(150) NOT NULL, qisqartma VARCHAR(10) UNIQUE, tashkil_yili YEAR, dekan_id INT, telefon VARCHAR(20), email VARCHAR(100) ); -- 4. KAFEDRA CREATE TABLE kafedra ( kafedra_id INT AUTO_INCREMENT PRIMARY KEY, nomi VARCHAR(150) NOT NULL, fakultet_id INT NOT NULL, mudiri_id INT, tashkil_yili YEAR, email VARCHAR(100), FOREIGN KEY (fakultet_id) REFERENCES fakultet(fakultet_id) ); -- 5. YONALISH CREATE TABLE yonalish ( yonalish_id INT AUTO_INCREMENT PRIMARY KEY, kodi VARCHAR(20) UNIQUE NOT NULL, nomi VARCHAR(200) NOT NULL, kafedra_id INT NOT NULL, daraja ENUM('bakalavr','magistr','doktorantura') NOT NULL, talim_shakli ENUM('kunduzgi','sirtqi') DEFAULT 'kunduzgi', davomiyligi TINYINT, kredit_soni INT, FOREIGN KEY (kafedra_id) REFERENCES kafedra(kafedra_id) ); -- 6. XODIM CREATE TABLE xodim ( xodim_id INT AUTO_INCREMENT PRIMARY KEY, xodim_raqam VARCHAR(20) UNIQUE, ism VARCHAR(80) NOT NULL, familiya VARCHAR(80) NOT NULL, otasining_ismi VARCHAR(80), kafedra_id INT, lavozim VARCHAR(100), ilmiy_daraja ENUM('none','magistr','phd','dsc') DEFAULT 'none', unvon ENUM('none','docent','professor') DEFAULT 'none', maosh DECIMAL(12,2), telefon VARCHAR(20), email VARCHAR(100), holati ENUM('aktiv','tatil','ketgan') DEFAULT 'aktiv', FOREIGN KEY (kafedra_id) REFERENCES kafedra(kafedra_id) ); -- FK (circular dependency yechimi) ALTER TABLE fakultet ADD CONSTRAINT fk_dekan FOREIGN KEY (dekan_id) REFERENCES xodim(xodim_id); ALTER TABLE kafedra ADD CONSTRAINT fk_mudiri FOREIGN KEY (mudiri_id) REFERENCES xodim(xodim_id); -- 7. GURUH CREATE TABLE guruh ( guruh_id INT AUTO_INCREMENT PRIMARY KEY, guruh_kodi VARCHAR(20) UNIQUE NOT NULL, yonalish_id INT NOT NULL, kurs TINYINT, akademik_yil VARCHAR(10), qoshiq_id INT, talabalar_soni INT DEFAULT 0, FOREIGN KEY (yonalish_id) REFERENCES yonalish(yonalish_id) ); -- 8. TALABA CREATE TABLE talaba ( talaba_id INT AUTO_INCREMENT PRIMARY KEY, talaba_raqam VARCHAR(20) UNIQUE NOT NULL, ism VARCHAR(80) NOT NULL, familiya VARCHAR(80) NOT NULL, otasining_ismi VARCHAR(80), tug_sana DATE, jins ENUM('erkak','ayol'), pasport_seria VARCHAR(15) UNIQUE, jshshir VARCHAR(14) UNIQUE, yonalish_id INT NOT NULL, kurs TINYINT, guruh_id INT, qabul_yili YEAR, holati ENUM('aktiv','tatil','chiqarilgan','bitirgan') DEFAULT 'aktiv', moliya_turi ENUM('grant','kontrakt') NOT NULL, telefon VARCHAR(20), email VARCHAR(100), yaratilgan TIMESTAMP DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (yonalish_id) REFERENCES yonalish(yonalish_id), FOREIGN KEY (guruh_id) REFERENCES guruh(guruh_id) ); -- Guruhning qoshiq talaba FK ALTER TABLE guruh ADD CONSTRAINT fk_qoshiq FOREIGN KEY (qoshiq_id) REFERENCES talaba(talaba_id); -- 9. FAN CREATE TABLE fan ( fan_id INT AUTO_INCREMENT PRIMARY KEY, kodi VARCHAR(20) UNIQUE NOT NULL, nomi VARCHAR(200) NOT NULL, kafedra_id INT NOT NULL, kredit TINYINT, ma_saat INT, amaliy_saat INT, lab_saat INT, semestr TINYINT, hisob_turi ENUM('imtihon','zachot') DEFAULT 'imtihon', FOREIGN KEY (kafedra_id) REFERENCES kafedra(kafedra_id) ); -- 10. DARS_JADVALI CREATE TABLE dars_jadvali ( jadval_id INT AUTO_INCREMENT PRIMARY KEY, fan_id INT NOT NULL, xodim_id INT NOT NULL, guruh_id INT NOT NULL, xona_id INT NOT NULL, kun ENUM('dushanba','seshanba','chorshanba','payshanba','juma','shanba'), para_raqam TINYINT, dars_turi ENUM('ma','amaliy','laboratoriya'), semestr TINYINT, FOREIGN KEY (fan_id) REFERENCES fan(fan_id), FOREIGN KEY (xodim_id) REFERENCES xodim(xodim_id), FOREIGN KEY (guruh_id) REFERENCES guruh(guruh_id), FOREIGN KEY (xona_id) REFERENCES xona(xona_id) ); -- 11. DAVOMAT CREATE TABLE davomat ( davomat_id INT AUTO_INCREMENT PRIMARY KEY, talaba_id INT NOT NULL, jadval_id INT NOT NULL, sana DATE NOT NULL, holat ENUM('keldi','kelmadi','kech_keldi','sababli') NOT NULL, izoh TEXT, UNIQUE KEY uq_davomat (talaba_id, jadval_id, sana), FOREIGN KEY (talaba_id) REFERENCES talaba(talaba_id), FOREIGN KEY (jadval_id) REFERENCES dars_jadvali(jadval_id) ); -- 12. IMTIHON CREATE TABLE imtihon ( imtihon_id INT AUTO_INCREMENT PRIMARY KEY, fan_id INT NOT NULL, guruh_id INT NOT NULL, xodim_id INT NOT NULL, xona_id INT, sana DATETIME NOT NULL, turi ENUM('imtihon','zachot','kursavoy'), semestr TINYINT, holati ENUM('rejalashtirilgan','otdi','bekor') DEFAULT 'rejalashtirilgan', FOREIGN KEY (fan_id) REFERENCES fan(fan_id), FOREIGN KEY (guruh_id) REFERENCES guruh(guruh_id), FOREIGN KEY (xodim_id) REFERENCES xodim(xodim_id), FOREIGN KEY (xona_id) REFERENCES xona(xona_id) ); -- 13. BAHO CREATE TABLE baho ( baho_id INT AUTO_INCREMENT PRIMARY KEY, talaba_id INT NOT NULL, fan_id INT NOT NULL, baholovchi_id INT, joriy_baho DECIMAL(5,2), oraliq_baho DECIMAL(5,2), yakuniy_baho DECIMAL(5,2), harf_baho ENUM('A','B','C','D','E','F'), gpa_ball DECIMAL(3,2), semestr TINYINT NOT NULL, sana DATETIME DEFAULT CURRENT_TIMESTAMP, UNIQUE KEY uq_baho (talaba_id, fan_id, semestr), FOREIGN KEY (talaba_id) REFERENCES talaba(talaba_id), FOREIGN KEY (fan_id) REFERENCES fan(fan_id), FOREIGN KEY (baholovchi_id) REFERENCES xodim(xodim_id) ); -- 14. DIPLOM_ISHI CREATE TABLE diplom_ishi ( diplom_id INT AUTO_INCREMENT PRIMARY KEY, talaba_id INT UNIQUE NOT NULL, mavzu TEXT NOT NULL, rahbar_id INT NOT NULL, topshirish_sanasi DATE, himoya_sanasi DATE, baho DECIMAL(5,2), holati ENUM('tayyorlanmoqda','topshirildi','himoya_qilindi') DEFAULT 'tayyorlanmoqda', FOREIGN KEY (talaba_id) REFERENCES talaba(talaba_id), FOREIGN KEY (rahbar_id) REFERENCES xodim(xodim_id) ); -- 15. TOLOV CREATE TABLE tolov ( tolov_id INT AUTO_INCREMENT PRIMARY KEY, talaba_id INT NOT NULL, miqdor DECIMAL(15,2) NOT NULL, turi ENUM('kontrakt','qayta_qabul','jarima'), usuli ENUM('naqd','bank','online'), sana DATETIME DEFAULT CURRENT_TIMESTAMP, kvitansiya VARCHAR(50), holati ENUM('kutilmoqda','tasdiqlandi','bekor') DEFAULT 'kutilmoqda', FOREIGN KEY (talaba_id) REFERENCES talaba(talaba_id) ); -- 16. STIPENDIYA CREATE TABLE stipendiya ( stip_id INT AUTO_INCREMENT PRIMARY KEY, talaba_id INT NOT NULL, turi ENUM('oddiy','nominal','davlat','prezident'), miqdor DECIMAL(12,2) NOT NULL, boshlanish DATE, tugash DATE, holati ENUM('aktiv','bekor','yakunlangan') DEFAULT 'aktiv', FOREIGN KEY (talaba_id) REFERENCES talaba(talaba_id) ); -- 17. KITOB CREATE TABLE kitob ( kitob_id INT AUTO_INCREMENT PRIMARY KEY, isbn VARCHAR(20) UNIQUE, nomi VARCHAR(255) NOT NULL, muallif VARCHAR(200), nashriyot VARCHAR(150), nashr_yili YEAR, turi ENUM('darslik','qollanma','ilmiy','badiiy'), jami_nusxa INT DEFAULT 1, mavjud_nusxa INT DEFAULT 1, fan_id INT, FOREIGN KEY (fan_id) REFERENCES fan(fan_id) ); -- 18. KITOB_IJARA CREATE TABLE kitob_ijara ( ijara_id INT AUTO_INCREMENT PRIMARY KEY, kitob_id INT NOT NULL, talaba_id INT NOT NULL, olish_sanasi DATE NOT NULL, qaytarish_sanasi DATE NOT NULL, qaytarildi DATE, holati ENUM('olindi','qaytarildi','kechiktirilgan') DEFAULT 'olindi', jarima DECIMAL(10,2) DEFAULT 0, FOREIGN KEY (kitob_id) REFERENCES kitob(kitob_id), FOREIGN KEY (talaba_id) REFERENCES talaba(talaba_id) ); -- ================================================ -- VIEW'LAR -- ================================================ -- Talabalar to'liq ma'lumot CREATE VIEW v_talabalar AS SELECT t.talaba_id, t.talaba_raqam, CONCAT(t.familiya,' ',t.ism) AS tolit_ismi, y.nomi AS yonalish, f.qisqartma AS fakultet, t.kurs, t.holati, t.moliya_turi FROM talaba t JOIN yonalish y ON t.yonalish_id = y.yonalish_id JOIN kafedra k ON y.kafedra_id = k.kafedra_id JOIN fakultet f ON k.fakultet_id = f.fakultet_id; -- GPA reytingi CREATE VIEW v_gpa_reyting AS SELECT t.talaba_raqam, CONCAT(t.familiya,' ',t.ism) AS ismi, ROUND(AVG(b.gpa_ball),2) AS gpa, COUNT(b.baho_id) AS fanlar_soni FROM talaba t LEFT JOIN baho b ON t.talaba_id = b.talaba_id GROUP BY t.talaba_id ORDER BY gpa DESC;
